
//Set icon item
setIcon(<twilightforest:castle_rune_brick:3>)

//Set title String
setTitle("Castle Rune Brick")

//Set description String
setDescription("Follow the advancements in the Twilight Forest tab to defeat the Twilight Forest and get a castle rune brick")

//Add a parent advancement. The syntax is "modid:folder/path/to/the/advancement/then/filename"
addParent("myconesia:magicmap")

//Set the toast notification in the top right corner of the screen to not show up when this advancement is achieved.
setShowToast(true)

//Set the advancement to not send a chat message when it is achieved.
setAnnounceToChat(false)

//Set the position
setPos(-82,91)

//Hide the connection lines, so I can have a pretty shape
//hideLines()
alwaysVisible()

//Adds criteria named "entered_forest" with the trigger type "minecraft:location". This function returns the criteria as an object
criteria = addCriteria("entered_highlands_center", "minecraft:location")

//Sets the required biome for the criteria. This is the biome the player must be in.
criteria.setBiome("twilightforest:highlands_center")
